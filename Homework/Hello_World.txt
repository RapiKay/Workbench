#include <stdio.h>            라이브러리를 추가하는 부분( 기본적인 함수를 제공하는 라이브러리를 불러옴 )

int main(void)                                ( 정수값은 int를 사용, 하나의 약속, void 는 매개변수 함수를 실행할 때 함수가 처리할 값을 받아오는 것, 대부분 int main(void)를 사용, 가장 첫 번째로 실행되는 함수 )
{
	printf("Hello World");            ( 사용자에게 Hello World 메세지를 출력해주는 기본적인 함수 )
	return 0;                           ( 하나의 함수가 종료될 때 반환값을 정해주는 지정된 약속 , 정수값을 리턴하는 int를 사용하여, 0이라는 정수를 return )
}

컴파일러가 실행시킬 수 있는 목적 프로그램으로 변환시킬 때 main함수부터 실행이 되도록 약속되어있다.
